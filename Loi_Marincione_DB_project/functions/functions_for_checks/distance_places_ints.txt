RETURN distance_places(
	( SELECT "Place".*::"Place" AS "Place"
	  FROM "Place"
	  WHERE ("Place"."ID" = distance_places_ints.code_a)),
	( SELECT "Place".*::"Place" AS "Place"
	  FROM "Place"
	  WHERE ("Place"."ID" = distance_places_ints.code_b))
);
