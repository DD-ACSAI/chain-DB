--  sits inside the DB and is handled by pgadmin
--  because one thing is a trigger function and another
--  is a trigger.

BEGIN
    IF NOT old."isExpired" AND new."isExpired" THEN
      UPDATE "Depot"
      SET "OccupiedPlaces" = "OccupiedPlaces" - 1
      WHERE "Depot"."PlaceCode" = old."PlaceCode";
    END IF;
    RETURN NULL;
END;
